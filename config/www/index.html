Verify the Demo is Working

# 1. Confirm scan data is flowing Gazebo → Nav2 via Zenoh
oc exec -n robot-demo deploy/robot-nav -c nav2 -- \
  bash -c 'source /usr/lib64/ros-jazzy/setup.bash; \
           timeout 15 ros2 topic echo /scan --once'

# 2. Check SLAM map size (transient_local QoS required — hz won't work)
oc exec -n robot-demo deploy/robot-nav -c nav2 -- \
  bash -c 'source /usr/lib64/ros-jazzy/setup.bash; \
           ros2 topic echo \
             --qos-reliability reliable \
             --qos-durability transient_local \
             /map --once 2>/dev/null | grep -E "frame_id|width|height|resolution"'

# 3. Check Nav2 is active and robot is navigating
oc exec -n robot-demo deploy/robot-nav -c nav2 -- \
  bash -c 'source /usr/lib64/ros-jazzy/setup.bash; \
           ros2 node list'

# 4. Check Zenoh bridge connectivity (topics visible on Nav2 pod)
oc exec -n robot-demo deploy/robot-nav -c nav2 -- \
  bash -c 'source /usr/lib64/ros-jazzy/setup.bash; \
           ros2 topic list'

# 5. Send a one-shot navigation goal
oc exec -n robot-demo deploy/robot-nav -c nav2 -- \
  bash -c 'source /usr/lib64/ros-jazzy/setup.bash; \
           ros2 topic pub -1 /goal_pose geometry_msgs/msg/PoseStamped \
             "{header:{frame_id:map},pose:{position:{x:1.0,y:1.0,z:0.0},orientation:{w:1.0}}}"'
